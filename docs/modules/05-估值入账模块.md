# 模块设计 - 估值入账模块

## 1. 模块概述

### 1.1 功能定位
估值入账模块采用多种估值方法（成本法、收益法、市场法）对数据资产进行价值评估，生成符合会计准则的入账方案，支持财务审批流程。

### 1.2 估值方法
1. **成本法**: 基于数据资产的开发、获取、维护成本
2. **收益法**: 基于数据资产未来产生的收益折现
3. **市场法**: 参考市场上类似数据资产的交易价格

## 2. 功能清单

| 功能点 | 优先级 | 说明 |
|--------|--------|------|
| 成本核算 | P0 | 计算数据资产成本 |

## 4. 估值引擎实现

### 4.1 成本法估值
```java
@Service
public class CostMethodValuation {
    
    /**
     * 成本法估值
     */
    public CostValuationResult calculate(
            DataProduct product,
            List<DataImportTask> importTasks,
            List<SystemAssessment> assessments) {
        
        CostValuationResult result = new CostValuationResult();
        
        // 1. 历史成本（已发生）
        BigDecimal historicalCost = calculateHistoricalCost(product);
        result.setHistoricalCost(historicalCost);
        
        // 2. 开发成本
        BigDecimal developmentCost = calculateDevelopmentCost(product);
        result.setDevelopmentCost(developmentCost);
        
        // 3. 获取成本（数据采购等）
        BigDecimal acquisitionCost = calculateAcquisitionCost(importTasks);
        result.setAcquisitionCost(acquisitionCost);
        
        // 4. 加工成本（清洗、转换等）
        BigDecimal processingCost = calculateProcessingCost(importTasks);
        result.setProcessingCost(processingCost);
        
        // 5. 存储成本
        BigDecimal storageCost = calculateStorageCost(product);
        result.setStorageCost(storageCost);
        
        // 6. 维护成本
        BigDecimal maintenanceCost = calculateMaintenanceCost(product);
        result.setMaintenanceCost(maintenanceCost);
        
        // 总成本
        BigDecimal totalCost = historicalCost
            .add(developmentCost)
            .add(acquisitionCost)
            .add(processingCost)
            .add(storageCost)
            .add(maintenanceCost);
        
        result.setTotalCost(totalCost);
        
        // 考虑成新率调整
        BigDecimal depreciationRate = calculateDepreciationRate(product);
        BigDecimal adjustedValue = totalCost.multiply(depreciationRate);
        
        result.setDepreciationRate(depreciationRate);
        result.setAdjustedValue(adjustedValue);
        
        return result;
    }
    
    private BigDecimal calculateHistoricalCost(DataProduct product) {
        // 获取历史投入成本
        return BigDecimal.valueOf(320000); // 示例
    }
    
    private BigDecimal calculateDevelopmentCost(DataProduct product) {
        // 根据产品复杂度估算开发成本
        int complexity = product.getComplexityScore(); // 1-10
        BigDecimal dailyRate = BigDecimal.valueOf(2000); // 人天成本
        int estimatedDays = complexity * 20; // 估算天数
        
        return dailyRate.multiply(BigDecimal.valueOf(estimatedDays));
    }
    
    private BigDecimal calculateStorageCost(DataProduct product) {
        // 根据数据量计算存储成本
        long dataSizeGB = product.getDataSizeGB();
        BigDecimal annualStorageCostPerGB = BigDecimal.valueOf(50); // 每GB每年成本
        int usefulLifeYears = 3; // 使用年限
        
        return annualStorageCostPerGB
            .multiply(BigDecimal.valueOf(dataSizeGB))
            .multiply(BigDecimal.valueOf(usefulLifeYears));
    }
    
    private BigDecimal calculateDepreciationRate(DataProduct product) {
        // 根据数据时效性计算成新率
        LocalDate createdDate = product.getCreatedAt().toLocalDate();
        long monthsSinceCreation = ChronoUnit.MONTHS.between(createdDate, LocalDate.now());
        
        // 假设数据价值每年折旧20%
        double depreciationRate = Math.max(0.2, 1 - (monthsSinceCreation * 0.0167));
        
        return BigDecimal.valueOf(depreciationRate);
    }
}
```

### 4.2 收益法估值
```java
@Service
public class IncomeMethodValuation {
    
    /**
     * 收益法估值 - DCF模型
     */
    public IncomeValuationResult calculate(
            DataProduct product,
            SystemAssessment assessment) {
        
        IncomeValuationResult result = new IncomeValuationResult();
        
        // 预测期（年）
        int forecastPeriod = 5;
        result.setForecastPeriodYears(forecastPeriod);
        
        // 折现率（WACC）
        BigDecimal discountRate = BigDecimal.valueOf(0.12); // 12%
        result.setDiscountRate(discountRate);
        
        // 预测现金流
        List<CashFlowProjection> projections = new ArrayList<>();
        BigDecimal totalNPV = BigDecimal.ZERO;
        
        BigDecimal baseBenefit = assessment.getEstimatedBenefit() != null ? 
            assessment.getEstimatedBenefit() : BigDecimal.valueOf(500000);
        
        BigDecimal growthRate = BigDecimal.valueOf(0.15); // 年增长率15%
        
        for (int year = 1; year <= forecastPeriod; year++) {
            CashFlowProjection projection = new CashFlowProjection();
            projection.setYear(year);
            
            // 收益增长
            BigDecimal benefit = baseBenefit
                .multiply(BigDecimal.ONE.add(growthRate).pow(year - 1));
            
            // 直接收益（60%）
            BigDecimal directRevenue = benefit.multiply(BigDecimal.valueOf(0.6));
            projection.setDirectRevenue(directRevenue);
            
            // 成本节约（25%）
            BigDecimal costSaving = benefit.multiply(BigDecimal.valueOf(0.25));
            projection.setCostSaving(costSaving);
            
            // 风险降低（10%）
            BigDecimal riskReduction = benefit.multiply(BigDecimal.valueOf(0.10));
            projection.setRiskReductionValue(riskReduction);
            
            // 决策支持（5%）
            BigDecimal decisionSupport = benefit.multiply(BigDecimal.valueOf(0.05));
            projection.setDecisionSupportValue(decisionSupport);
            
            // 总收益
            BigDecimal totalBenefit = directRevenue
                .add(costSaving)
                .add(riskReduction)
                .add(decisionSupport);
            projection.setTotalBenefit(totalBenefit);
            
            // 折现
            BigDecimal discountFactor = BigDecimal.ONE
                .add(discountRate)
                .pow(year, MathContext.DECIMAL64);
            BigDecimal presentValue = totalBenefit.divide(
                discountFactor, 2, RoundingMode.HALF_UP);
            projection.setPresentValue(presentValue);
            
            projections.add(projection);
            totalNPV = totalNPV.add(presentValue);
        }
        
        result.setCashFlowProjections(projections);
        result.setNetPresentValue(totalNPV);
        result.setIncomeMethodValue(totalNPV);
        
        // 计算IRR（简化计算）
        double irr = calculateIRR(projections);
        result.setInternalRateOfReturn(BigDecimal.valueOf(irr));
        
        // 计算投资回收期
        int paybackPeriod = calculatePaybackPeriod(projections);
        result.setPaybackPeriodMonths(paybackPeriod);
        
        return result;
    }
    
    private double calculateIRR(List<CashFlowProjection> projections) {
        // 使用牛顿迭代法计算IRR
        double irr = 0.15; // 初始猜测值
        double precision = 0.0001;
        int maxIterations = 100;
        
        for (int i = 0; i < maxIterations; i++) {
            double npv = 0;
            double derivative = 0;
            
            for (CashFlowProjection proj : projections) {
                double cashFlow = proj.getTotalBenefit().doubleValue();
                int year = proj.getYear();
                
                npv += cashFlow / Math.pow(1 + irr, year);
                derivative -= year * cashFlow / Math.pow(1 + irr, year + 1);
            }
            
            double newIRR = irr - npv / derivative;
            
            if (Math.abs(newIRR - irr) < precision) {
                return newIRR;
            }
            
            irr = newIRR;
        }
        
        return irr;
    }
}
```

### 4.3 综合估值计算
```java
@Service
public class ComprehensiveValuationService {
    
    @Autowired
    private CostMethodValuation costMethod;
    
    @Autowired
    private IncomeMethodValuation incomeMethod;
    
    @Autowired
    private MarketMethodValuation marketMethod;
    
    /**
     * 执行综合估值
     */
    public ValuationResult performValuation(Long dataProductId) {
        DataProduct product = dataProductRepository.findById(dataProductId)
            .orElseThrow(() -> new BusinessException("数据产品不存在"));
        
        ValuationResult result = new ValuationResult();
        result.setDataProductId(dataProductId);
        
        // 1. 成本法估值
        CostValuationResult costResult = costMethod.calculate(
            product, 
            getImportTasks(product),
            getAssessments(product)
        );
        result.setCostMethodResult(costResult);
        result.setCostMethodValue(costResult.getAdjustedValue());
        
        // 2. 收益法估值
        IncomeValuationResult incomeResult = incomeMethod.calculate(
            product,
            getSystemAssessment(product)
        );
        result.setIncomeMethodResult(incomeResult);
        result.setIncomeMethodValue(incomeResult.getIncomeMethodValue());
        
        // 3. 市场法估值（如有可比交易）
        MarketValuationResult marketResult = marketMethod.calculate(product);
        result.setMarketMethodResult(marketResult);
        result.setMarketMethodValue(marketResult.getMarketValue());
        
        // 4. 确定权重（根据数据特点）
        ValuationWeights weights = determineWeights(product);
        result.setValuationWeights(weights);
        
        // 5. 计算加权估值
        BigDecimal weightedValue = 
            result.getCostMethodValue().multiply(weights.getCostWeight())
                .add(result.getIncomeMethodValue().multiply(weights.getIncomeWeight()))
                .add(result.getMarketMethodValue().multiply(weights.getMarketWeight()));
        
        result.setWeightedValuation(weightedValue);
        
        // 6. 应用调整
        List<ValuationAdjustment> adjustments = calculateAdjustments(product);
        BigDecimal finalAdjustment = adjustments.stream()
            .map(ValuationAdjustment::getAmount)
            .reduce(BigDecimal.ZERO, BigDecimal::add);
        
        BigDecimal finalValuation = weightedValue.add(finalAdjustment);
        result.setFinalValuation(finalValuation);
        result.setAdjustments(adjustments);
        
        // 7. 计算财务指标
        result.setNetPresentValue(incomeResult.getNetPresentValue());
        result.setInternalRateOfReturn(incomeResult.getInternalRateOfReturn());
        result.setPaybackPeriodMonths(incomeResult.getPaybackPeriodMonths());
        result.setReturnOnInvestment(
            calculateROI(costResult.getTotalCost(), finalValuation)
        );
        
        return result;
    }
    
    private ValuationWeights determineWeights(DataProduct product) {
        ValuationWeights weights = new ValuationWeights();
        
        // 根据数据产品特点确定权重
        String productType = product.getProductType();
        
        if ("decision_support".equals(productType)) {
            // 决策支持类产品，收益法权重更高
            weights.setCostWeight(BigDecimal.valueOf(0.20));
            weights.setIncomeWeight(BigDecimal.valueOf(0.60));
            weights.setMarketWeight(BigDecimal.valueOf(0.20));
        } else if ("operational".equals(productType)) {
            // 运营类产品，成本法权重更高
            weights.setCostWeight(BigDecimal.valueOf(0.40));
            weights.setIncomeWeight(BigDecimal.valueOf(0.40));
            weights.setMarketWeight(BigDecimal.valueOf(0.20));
        } else {
            // 分析类产品，均衡权重
            weights.setCostWeight(BigDecimal.valueOf(0.30));
            weights.setIncomeWeight(BigDecimal.valueOf(0.50));
            weights.setMarketWeight(BigDecimal.valueOf(0.20));
        }
        
        return weights;
    }
}
```
