# 模块设计 - 数据导入与产品模块

## 1. 模块概述

### 1.1 功能定位
数据导入模块负责将客户的数据文件（CSV、Excel等）导入系统，进行字段映射、数据清洗和数据产品字段筛选；数据产品模块负责定义和设计最终输出的数据产品。

### 1.2 核心流程
```
文件上传 → 格式检测 → 数据预览 → 字段映射 → 数据清洗 → 
产品字段筛选 → 质量检查 → 数据入库 → 产品生成
```

### 2.3 核心算法实现

#### 2.3.1 智能字段映射
```java
@Service
public class SmartFieldMapper {
    
    @Autowired
    private FieldMappingTemplateRepository templateRepository;
    
    /**
     * 推荐字段映射
     */
    public List<MappingSuggestion> suggestMappings(
            List<String> sourceFields,
            String systemType,
            Long customerId) {
        
        List<MappingSuggestion> suggestions = new ArrayList<>();
        
        // 1. 加载映射模板
        FieldMappingTemplate template = templateRepository
            .findBySystemType(systemType);
        
        // 2. 为每个源字段寻找最佳匹配
        for (String sourceField : sourceFields) {
            MappingSuggestion suggestion = new MappingSuggestion();
            suggestion.setSourceField(sourceField);
            
            // 2.1 精确匹配
            MappingRule exactMatch = findExactMatch(sourceField, template);
            if (exactMatch != null) {
                suggestion.setTargetField(exactMatch.getStandardField());
                suggestion.setConfidence(1.0);
                suggestion.setMatchType("exact");
                suggestion.setDataType(exactMatch.getDataType());
            } else {
                // 2.2 模糊匹配
                List<MappingScore> candidates = calculateSimilarity(
                    sourceField, 
                    template.getAllStandardFields()
                );
                
                if (!candidates.isEmpty() && candidates.get(0).getScore() > 0.6) {
                    MappingScore best = candidates.get(0);
                    suggestion.setTargetField(best.getField());
                    suggestion.setConfidence(best.getScore());
                    suggestion.setMatchType("fuzzy");
                    suggestion.setDataType(inferDataType(sourceField));
                } else {
                    // 2.3 无匹配，建议新建
                    suggestion.setTargetField(null);
                    suggestion.setConfidence(0.0);
                    suggestion.setMatchType("new");
                    suggestion.setSuggestedDataType(inferDataType(sourceField));
                }
            }
            
            suggestions.add(suggestion);
        }
        
        return suggestions;
    }
    
    /**
     * 计算字段相似度（综合算法）
     */
    private double calculateSimilarity(String source, String target) {
        // 1. 编辑距离相似度
        double editSim = calculateEditDistanceSimilarity(source, target);
        
        // 2. 分词后Jaccard相似度（中文）
        double jaccardSim = calculateJaccardSimilarity(source, target);
        
        // 3. 拼音相似度（中文）
        double pinyinSim = calculatePinyinSimilarity(source, target);
        
        // 4. 关键词匹配
        double keywordSim = calculateKeywordMatch(source, target);
        
        // 加权平均
        return editSim * 0.3 + jaccardSim * 0.3 + pinyinSim * 0.2 + keywordSim * 0.2;
    }
    
    /**
     * 推断数据类型
     */
    private String inferDataType(String fieldName, List<String> sampleValues) {
        // 基于字段名推断
        String lowerName = fieldName.toLowerCase();
        
        if (lowerName.contains("id") || lowerName.contains("code") || 
            lowerName.contains("no") || lowerName.contains("编号")) {
            return "STRING";
        }
        
        if (lowerName.contains("date") || lowerName.contains("time") ||
            lowerName.contains("日期") || lowerName.contains("时间")) {
            return "DATETIME";
        }
        
        if (lowerName.contains("amount") || lowerName.contains("price") ||
            lowerName.contains("money") || lowerName.contains("金额") ||
            lowerName.contains("价格")) {
            return "DECIMAL";
        }
        
        if (lowerName.contains("count") || lowerName.contains("num") ||
            lowerName.contains("qty") || lowerName.contains("数量")) {
            return "INTEGER";
        }
        
        // 基于样本值推断
        if (sampleValues != null && !sampleValues.isEmpty()) {
            return inferTypeFromSamples(sampleValues);
        }
        
        return "STRING";
    }
}
```

#### 2.3.2 数据清洗引擎
```java
@Service
public class DataCleaningEngine {
    
    /**
     * 执行数据清洗
     */
    public CleaningResult cleanData(
            List<Map<String, Object>> rawData,
            List<CleaningRule> rules) {
        
        CleaningResult result = new CleaningResult();
        List<Map<String, Object>> cleanedData = new ArrayList<>();
        List<CleaningError> errors = new ArrayList<>();
        
        int rowNum = 0;
        for (Map<String, Object> row : rawData) {
            rowNum++;
            Map<String, Object> cleanedRow = new HashMap<>(row);
            boolean hasError = false;
            
            for (CleaningRule rule : rules) {
                String field = rule.getField();
                Object value = row.get(field);
                
                try {
                    Object cleanedValue = applyRule(value, rule);
                    cleanedRow.put(field, cleanedValue);
                } catch (CleaningException e) {
                    hasError = true;
                    errors.add(new CleaningError(rowNum, field, e.getMessage()));
                    
                    if (rule.getErrorHandling() == ErrorHandling.SKIP_ROW) {
                        cleanedRow = null;
                        break;
                    } else if (rule.getErrorHandling() == ErrorHandling.USE_DEFAULT) {
                        cleanedRow.put(field, rule.getDefaultValue());
                    }
                }
            }
            
            if (cleanedRow != null) {
                cleanedData.add(cleanedRow);
            }
            
            if (hasError) {
                result.incrementErrorCount();
            }
        }
        
        result.setCleanedData(cleanedData);
        result.setErrors(errors);
        result.setTotalRows(rawData.size());
        result.setCleanedRows(cleanedData.size());
        
        return result;
    }
    
    /**
     * 应用清洗规则
     */
    private Object applyRule(Object value, CleaningRule rule) {
        if (value == null) {
            if (rule.isRequired()) {
                throw new CleaningException("必填字段为空");
            }
            return rule.getDefaultValue();
        }
        
        switch (rule.getRuleType()) {
            case TRIM:
                return value.toString().trim();
                
            case UPPERCASE:
                return value.toString().toUpperCase();
                
            case LOWERCASE:
                return value.toString().toLowerCase();
                
            case REGEX_REPLACE:
                return value.toString().replaceAll(
                    rule.getPattern(), 
                    rule.getReplacement()
                );
                
            case DATE_PARSE:
                return parseDate(value.toString(), rule.getDateFormat());
                
            case NUMBER_PARSE:
                return parseNumber(value.toString());
                
            case ENUM_MAPPING:
                return mapEnum(value.toString(), rule.getEnumMappings());
                
            case LENGTH_LIMIT:
                String str = value.toString();
                if (str.length() > rule.getMaxLength()) {
                    return str.substring(0, rule.getMaxLength());
                }
                return str;
                
            default:
                return value;
        }
    }
}
```

### 3.2 产品字段筛选算法

```java
@Service
public class ProductFieldSelector {
    
    /**
     * 根据产品模板筛选字段
     */
    public FieldSelectionResult selectFields(
            DataProductTemplate template,
            List<FieldMetadata> availableFields,
            List<Map<String, Object>> sampleData) {
        
        FieldSelectionResult result = new FieldSelectionResult();
        List<SelectedField> selectedFields = new ArrayList<>();
        
        // 1. 匹配必需字段
        for (ProductFieldRequirement requirement : template.getRequiredFields()) {
            FieldMatch match = findBestMatch(requirement, availableFields, sampleData);
            
            if (match.getConfidence() >= 0.7) {
                selectedFields.add(new SelectedField()
                    .setSourceField(match.getSourceField())
                    .setTargetField(requirement.getFieldName())
                    .setConfidence(match.getConfidence())
                    .setRequired(true)
                    .setMatchReason("必需字段匹配")
                );
            } else {
                result.addMissingRequiredField(requirement.getFieldName());
            }
        }
        
        // 2. 匹配可选字段
        for (ProductFieldRequirement requirement : template.getOptionalFields()) {
            FieldMatch match = findBestMatch(requirement, availableFields, sampleData);
            
            if (match.getConfidence() >= 0.8) {
                selectedFields.add(new SelectedField()
                    .setSourceField(match.getSourceField())
                    .setTargetField(requirement.getFieldName())
                    .setConfidence(match.getConfidence())
                    .setRequired(false)
                    .setMatchReason("高置信度可选字段")
                );
            }
        }
        
        // 3. 添加业务价值字段（额外）
        List<FieldMetadata> businessValueFields = identifyBusinessValueFields(availableFields);
        for (FieldMetadata field : businessValueFields) {
            if (!isAlreadySelected(field, selectedFields)) {
                selectedFields.add(new SelectedField()
                    .setSourceField(field.getName())
                    .setTargetField(field.getName())
                    .setConfidence(1.0)
                    .setRequired(false)
                    .setMatchReason("高业务价值字段")
                );
            }
        }
        
        result.setSelectedFields(selectedFields);
        result.calculateCoverage();
        
        return result;
    }
    
    /**
     * 识别高业务价值字段
     */
    private List<FieldMetadata> identifyBusinessValueFields(List<FieldMetadata> fields) {
        Set<String> businessKeywords = Set.of(
            "amount", "price", "cost", "revenue", "profit",
            "金额", "价格", "成本", "收入", "利润",
            "customer", "client", "客户",
            "order", "transaction", "订单", "交易",
            "product", "商品", "产品",
            "quantity", "count", "数量"
        );
        
        return fields.stream()
            .filter(field -> {
                String name = field.getName().toLowerCase();
                return businessKeywords.stream().anyMatch(name::contains);
            })
            .sorted((a, b) -> {
                // 按数据质量排序
                return Double.compare(b.getQualityScore(), a.getQualityScore());
            })
            .limit(20) // 最多选择20个
            .collect(Collectors.toList());
    }
}
```
